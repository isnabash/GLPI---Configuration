# GLPI---Configuration
Installation and Configuration of GLPI 

1st: INSTALLATION AND CONFIGURATION OF nginx

$ sudo apt update && sudo apt upgrade -y
$ sudo apt install nginx -y
$ sudo systemctl start nginx
$ sudo systemctl enable nginx
$ sudo systemctl status nginx

2st: CHECK THE FIREWALL STATUS

$ sudo ufw status -- if appears inactive, don`t do nothing; if not, run this command
$ sudo ufw allow 80
$ sudo ufw allow 443

3st: Open a web browser and verify with your IP address if webserver is working properly. 

4st: INSTALL & CONFIGURE DATABASE MARIDA-DB

 sudo apt install mariadb-server mariadb-client -y
$ sudo sytemctl enable mariadb 
$ sudo systemctl start mariadb
$ sudo systemctl status mariadb
$ sudo mysql_secure_installation (in the follow steps, just press ENTER until the prgrm ask to define password and then enter your password and press ENTER to continue)
$ sudo mysql -uroot -p
  > show databases;

# Access GLPI site to download the latest version (https://github.com/glpi-project/glpi/releases/download/11.0.6/glpi-11.0.6.tgz)

cd /var/www/$
/var/www/$ wget https://github.com/glpi-project/glpi/releases/download/11.0.6/glpi-11.0.6.tgz
/var/www/$ ls
/var/www/$ tar -vxf glpi-11.0.6.tgz
/var/www/$ sudo chown -R www-data:www-data glpi
/var/www/$ sudo chmod -R 755 glpi
/var/www/$ ls -l or ll
/var/www/$ sudo vim /etc/nginx/sites-available/default - (replace all the configuration with the new one)

server {
        listen 80 default_server;
        listen [::]:80 default_server;

        server_name _;

        root /var/www/glpi/public;
        index index.php;

        location / {
                try_files $uri /index.php$is_args$args;
        }

        location ~ ^/index\.php$ {
                include fastcgi_params;
                fastcgi_pass unix:/run/php/php8.3-fpm.sock;
                fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        }
}

/var/www/$ sudo nginx -t
/var/www/$ sudo systemctl reload nginx
/var/www/$ ifconfig
